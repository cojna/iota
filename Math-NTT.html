ntt :: forall p. KnownNat p => Vector (GF p) -> Vector (GF p) Source #

Number Theoretic Transform p: prime (c * 2 ^ k + 1) n = 2 ^ i, n < 2 ^ k

O(n log n) >>> ntt 998244353 [1,1,1,1] [4,0,0,0] >>> ntt 469762049 [123,0,0,0] [123,123,123,123]
